表注释:
-- 查询所有表注释
SELECT tb.table_name, d.description
FROM information_schema.tables tb
         JOIN pg_class c ON c.relname = tb.table_name
         LEFT JOIN pg_description d ON d.objoid = c.oid AND d.objsubid = '0'
WHERE tb.table_schema = 'test_schema';


-- 查询所有列
SELECT c.relname , a.attname,d.description,t.typname  ,n.nspname --c.relnamespace -- ,ts.spcname 
FROM  pg_class c --LEFT JOIN pg_tablespace ts ON c.reltablespace = ts.oid
LEFT JOIN pg_namespace n ON c.relnamespace = n.oid
,pg_attribute a, pg_type t, pg_description d -- pg_database da
where a.attrelid=c.oid and a.atttypid=t.oid and d.objoid=a.attrelid and d.objsubid=a.attnum 
--AND  c.relname = upper('t_110000_result')
-- AND c.relnamespace = da.dattablespace 
AND d.description LIKE '%服务%'
AND c.relname IN (lower('T_CAOT_TASK_EXTD_NEW'),lower('T_CAOT_ADV_TASK'))
--AND a.attname LIKE  lower('%QUES_DESC%')
--AND a.attname = 'select_id'
--AND a.attname LIKE lower('%ext_key12_base%')
;

join pg_user pu on pd.datdba = pu.usesysid

预估数据量:
select n_live_tup as estimate_rows from pg_stat_all_tables where relname = 'aken01';

 select reltuples::bigint as estimate_rows from pg_class where relname = 'aken01'; 

select (reltuples/relpages) * (pg_relation_size('aken01')/(current_setting('block_size')::integer)) as rows from pg_class where relname = 'aken01';

查询表及其索引使用情况
SELECT 
    relname AS table_name, 
    pg_size_pretty(pg_total_relation_size(relid)) AS total_size
FROM 
    pg_catalog.pg_statio_user_tables
ORDER BY 
    pg_total_relation_size(relid) DESC;
